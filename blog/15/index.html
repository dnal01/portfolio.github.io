Promises

Promises Syntaxis


const timeOut = (t) => {
return new Promise((resolve, reject) => {
    setTimeout(() => {
    resolve(`Completed in ${t}`)
    }, t)
})
}

// Resolving a normal promise.
timeOut(1000)
.then(result => console.log(result)) // Completed in 1000

// Promise.all
Promise.all([timeOut(1000), timeOut(2000)])
.then(result => console.log(result)) // ["Completed in 1000", "Completed in 2000"]

Promise.all([oneCall to Promise, secondCall to Promise]); waits until all Promises are resolved or rejected.
