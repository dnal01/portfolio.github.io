MySQL

MySQL requires a server. You can create a server by installing MySQL on your machine. When the server is running you need to connect your data base to it.

MySQL commands

CREATE DATABASE db_name;

CREATE USER 'user_name'@'localhost'
IDENTIFIED BY 'password';

GRANT CREATE, ALTER, DROP
ON demo_db.*
TO 'dev'@'localhost';

GRANT INSERT, SELECT, UPDATE, DELETE
ON demo_db.*
TO 'dev'@'localhost';

GRANT REFERENCES
ON demo_db.*
TO 'dev'@'localhost';

GRANT CREATE VIEW
ON demo_db.*
TO 'dev'@'localhost';

// Login to MySQL
...

// To start working with the dataBase specify it
USE db_name;

CREATE TABLE IF NOT EXISTS table_name
(
    column_name data_type NOT NULL AUTO_INCREMENT,
    ...,
    PRIMARY KEY (column_name)
)
